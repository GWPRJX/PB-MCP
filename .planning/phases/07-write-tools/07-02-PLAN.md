  ```typescript
    });
  });
  ```

  **2. Tool registration tests:**
  Test that `registerWriteTools` registers all 6 tools when no filter is provided, and that it respects the filter Set when provided.

  Use `vi.fn()` to mock the McpServer's `tool()` method and capture registration calls:
  ```typescript
  describe('registerWriteTools', () => {
    it('registers all 6 write tools when no filter is set', () => {
      const mockTool = vi.fn();
      const mockServer = { tool: mockTool } as unknown as McpServer;
      registerWriteTools(mockServer, null);
      expect(mockTool).toHaveBeenCalledTimes(6);
      // Verify each tool name is registered
      const registeredNames = mockTool.mock.calls.map((c: any[]) => c[0]);
      expect(registeredNames).toContain('create_stock_entry');
      expect(registeredNames).toContain('update_stock_entry');
      expect(registeredNames).toContain('create_invoice');
      expect(registeredNames).toContain('update_invoice');
      expect(registeredNames).toContain('create_contact');
      expect(registeredNames).toContain('update_contact');
    });

    it('filters tools based on enabled set', () => {
      const mockTool = vi.fn();
      const mockServer = { tool: mockTool } as unknown as McpServer;
      const filter = new Set(['create_contact', 'update_contact']);
      registerWriteTools(mockServer, filter);
      expect(mockTool).toHaveBeenCalledTimes(2);
      const registeredNames = mockTool.mock.calls.map((c: any[]) => c[0]);
      expect(registeredNames).toContain('create_contact');
      expect(registeredNames).toContain('update_contact');
      expect(registeredNames).not.toContain('create_invoice');
    });

    it('registers zero tools when filter excludes all write tools', () => {
      const mockTool = vi.fn();
      const mockServer = { tool: mockTool } as unknown as McpServer;
      const filter = new Set(['list_products']); // only a read tool
      registerWriteTools(mockServer, filter);
      expect(mockTool).toHaveBeenCalledTimes(0);
    });
  });
  ```

  **3. MCP server integration test:**
  Test that `createMcpServer()` includes write tools when they're in the enabled list, and excludes them when they're not:
  ```typescript
  describe('createMcpServer with write tools', () => {
    it('includes write tools when enabled', () => {
      const server = createMcpServer(['create_contact', 'list_products']);
      // Server should be created without errors
      expect(server).toBeDefined();
    });

    it('creates server with all tools when no filter', () => {
      const server = createMcpServer();
      expect(server).toBeDefined();
    });
  });
  ```

  **4. Tool description tests:**
  Verify each registered tool has a non-empty description and a Zod schema:
  ```typescript
  it('each write tool has a description and schema', () => {
    const mockTool = vi.fn();
    const mockServer = { tool: mockTool } as unknown as McpServer;
    registerWriteTools(mockServer, null);
    for (const call of mockTool.mock.calls) {
      const [name, description, schema, handler] = call;
      expect(typeof name).toBe('string');
      expect(typeof description).toBe('string');
      expect(description.length).toBeGreaterThan(10);
      expect(schema).toBeDefined();
      expect(typeof handler).toBe('function');
    }
  });
  ```

  Keep tests focused on integration points — tool registration, filtering, ALL_TOOLS membership. Do NOT test live POSibolt API calls (those require a running POSibolt instance).
