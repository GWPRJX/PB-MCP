**src/kb/sync.ts** — YouTrack fetch + atomic write-then-swap:

```typescript
import { createHash } from 'crypto';
import { sql } from '../db/client.js';

export interface SyncResult {
  article_count: number;
  synced_at: Date;
}

interface YouTrackArticle {
  id: string;
  idReadable: string;
  summary: string;
  content?: string;
  tags?: Array<{ name: string }>;
}

/**
 * Fetch all KB articles from YouTrack using paginated $top/$skip loop,
 * then atomically replace all rows in kb_articles (DELETE + INSERT in one transaction).
 *
 * KB-01: pulls articles from YouTrack REST API
 * KB-02: uses $top/$skip pagination loop
 * KB-07: single transaction ensures partial failures leave existing cache intact
 *
 * Returns { article_count, synced_at } on success.
 * Returns { article_count: 0, synced_at: now } and logs warning if env vars absent.
 */
export async function syncKbArticles(): Promise<SyncResult> {
  const baseUrl = process.env.YOUTRACK_BASE_URL;
  const token = process.env.YOUTRACK_TOKEN;
  const project = process.env.YOUTRACK_PROJECT ?? 'P8';

  if (!baseUrl || !token) {
    process.stderr.write('[kb/sync] WARNING: YOUTRACK_BASE_URL or YOUTRACK_TOKEN not set — skipping sync\n');
    return { article_count: 0, synced_at: new Date() };
  }

  const articles: YouTrackArticle[] = [];
  const pageSize = 100;
  let skip = 0;

  // Paginated fetch loop — loop until a page returns fewer items than pageSize
  while (true) {
    const fields = 'id,idReadable,summary,content,tags(name),parentArticle(id,idReadable,summary)';
    const url = `${baseUrl}/api/articles?fields=${fields}&query=project:${project}&$top=${pageSize}&$skip=${skip}`;

    const res = await fetch(url, {
      headers: {
        'Authorization': `Bearer ${token}`,
        'Accept': 'application/json',
      },
    });

    if (!res.ok) {
      const text = await res.text();
      throw new Error(`[kb/sync] YouTrack API error ${res.status}: ${text}`);
    }

    const page: YouTrackArticle[] = await res.json() as YouTrackArticle[];
    articles.push(...page);

    if (page.length < pageSize) {
      break; // last page
    }
    skip += pageSize;
  }

  process.stderr.write(`[kb/sync] Fetched ${articles.length} articles from YouTrack project:${project}\n`);

  const syncedAt = new Date();

  // Atomic write-then-swap: DELETE all + INSERT all in a single transaction (KB-07).
  // On error the transaction rolls back — existing cache rows are untouched.
  await sql.begin(async (tx) => {
    await tx`DELETE FROM kb_articles`;

    if (articles.length > 0) {
      // Build values for bulk insert
      const rows = articles.map((a) => {
        const tags = (a.tags ?? []).map((t) => t.name);
        const contentHash = a.content
          ? createHash('sha256').update(a.content).digest('hex')
          : null;
        return {
          youtrack_id: a.idReadable,
          summary: a.summary,
          content: a.content ?? null,
          tags,
          synced_at: syncedAt,
          content_hash: contentHash,
        };
      });

      // postgres.js supports inserting an array of objects with tagged template
      for (const row of rows) {
        await tx`
          INSERT INTO kb_articles (youtrack_id, summary, content, tags, synced_at, content_hash)
          VALUES (
            ${row.youtrack_id},
            ${row.summary},
            ${row.content},
            ${tx.array(row.tags)},
            ${row.synced_at},
            ${row.content_hash}
          )
        `;
      }
    }
  });

  process.stderr.write(`[kb/sync] Sync complete: ${articles.length} articles stored\n`);
  return { article_count: articles.length, synced_at: syncedAt };
}
```

**src/kb/scheduler.ts** — setInterval wrapper (KB-03, KB-05):

```typescript
import { syncKbArticles } from './sync.js';

/**
 * Start the KB auto-sync scheduler.
 *
 * KB-03: Runs syncKbArticles() on a configurable interval.
 * KB_SYNC_INTERVAL_MS defaults to 30 minutes (1_800_000 ms).
 *
 * Call this once at server startup (src/index.ts).
 * Do NOT call in tests — it would interfere with test isolation.
 *
 * Returns the interval handle so callers can clearInterval() on shutdown.
 */
export function startKbScheduler(): ReturnType<typeof setInterval> {
  const intervalMs = parseInt(process.env.KB_SYNC_INTERVAL_MS ?? '1800000', 10);

  process.stderr.write(`[kb/scheduler] Starting KB auto-sync every ${intervalMs}ms\n`);

  // Run once immediately at startup (non-blocking — don't await)
  syncKbArticles().catch((err) => {
    process.stderr.write(`[kb/scheduler] Initial sync failed: ${err instanceof Error ? err.message : String(err)}\n`);
  });

  // Then repeat on interval
  return setInterval(() => {
    syncKbArticles().catch((err) => {
      process.stderr.write(`[kb/scheduler] Scheduled sync failed: ${err instanceof Error ? err.message : String(err)}\n`);
    });
  }, intervalMs);
}
```

```typescript
import postgres from 'postgres';

const cleanSql = postgres(process.env.DATABASE_MIGRATION_URL ?? 'postgres://postgres:postgres@localhost:5432/pb_mcp');

    text: async () => '',

beforeEach(async () => {
  // Clean slate before each test
  await cleanSql`TRUNCATE TABLE kb_articles`;

// Close pool after all tests
import { afterAll } from 'vitest';
afterAll(async () => {
  await cleanSql.end();
});

describe('syncKbArticles()', () => {
```
